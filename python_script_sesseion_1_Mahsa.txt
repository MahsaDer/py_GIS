path=open(r'C:\Users\mahsa.derakhshan\Documents\GitHub\py_GIS\py_GIS\The Evolution of Smart Cities.txt')
book=path.read()
path.close()

text=book.split()

words={}
for word in text:
    word=word.lower().strip(",.?!;:")
    if word in words:
        words[word]+=1
    else:
        words[word]=1
#print(words)


repeated_words={}
for word, count in words.items():
    if count > 40:
        repeated_words[word]=count
print(repeated_words)
